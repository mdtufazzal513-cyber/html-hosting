firebase.initializeApp(firebaseConfig); const db = firebase.firestore(); // Auto-Save Logic const editor = document.getElementById('code'); window.onload = () => { if(localStorage.getItem('draft')) editor.value = localStorage.getItem('draft'); }; editor.addEventListener('input', () => localStorage.setItem('draft', editor.value)); function clearEditor() { if(confirm("Clear everything?")) { editor.value = ''; localStorage.removeItem('draft'); } } function showToast(msg) { const x = document.getElementById("toast"); x.innerText = msg; x.className = "show"; setTimeout(() => { x.className = x.className.replace("show", ""); }, 3000); } async function publish() { const html = editor.value; const customId = document.getElementById('customId').value.trim().replace(/[^a-zA-Z0-9-_]/g, ''); const pin = document.getElementById('pin').value.trim(); // 1. Validations if (!html.trim()) return showToast("⚠️ কোড খালি রাখা যাবে না!"); if (!pin) return showToast("⚠️ একটি পিন সেট করুন (ভবিষ্যতে ডিলিট করার জন্য)"); // 2. Banned Words Filter const banned = ['hacked', 'hacked by', 'scam', 'phishing', 'login password']; if (banned.some(word => html.toLowerCase().includes(word))) { return showToast("⛔ আপনার কোডে নিষিদ্ধ শব্দ আছে!"); } const btn = document.getElementById('pubBtn'); btn.innerHTML = ' Checking...'; btn.disabled = true; try { // 3. ID Generation let siteId = customId ? customId : Math.random().toString(36).substring(2, 10); // Check availability const docRef = db.collection('sites').doc(siteId); const docSnap = await docRef.get(); if (docSnap.exists) { showToast("⚠️ এই নামটি আগেই নেওয়া হয়েছে, অন্য নাম দিন।"); btn.innerHTML = ' Publish Now'; btn.disabled = false; return; } // 4. Save to Database await docRef.set({ html: html, pin: pin, views: 0, reportCount: 0, createdAt: firebase.firestore.FieldValue.serverTimestamp(), status: 'active' }); // Success localStorage.removeItem('draft'); const url = window.location.origin + window.location.pathname.replace('index.html', '') + 'view.html?id=' + siteId; document.getElementById('finalUrl').innerText = url; document.getElementById('modal').style.display = "flex"; } catch (err) { showToast("Error: " + err.message); } finally { btn.innerHTML = ' Publish Now'; btn.disabled = false; } } function copyLink() { navigator.clipboard.writeText(document.getElementById('finalUrl').innerText); showToast("Link Copied!"); } function closeModal() { document.getElementById('modal').style.display = "none"; } function share(platform) { const url = encodeURIComponent(document.getElementById('finalUrl').innerText); if(platform === 'wa') window.open(`https://wa.me/?text=${url}`); if(platform === 'fb') window.open(`https://www.facebook.com/sharer/sharer.php?u=${url}`); }
